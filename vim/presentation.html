Define Function

[!] Quietly replace existing function .

 1 function! CmdLine(str)
 2     exe "menu Foo.Bar :" . a:str
 3     emenu Foo.Bar
 4     unmenu Foo
 5 endfunction
 6 
 7 " From an idea by Michael Naumann
 8 function! VisualSearch(direction) range
 9     let l:saved_reg = @"
10     execute "normal! vgvy"
11 
12     let l:pattern = escape(@", '\\/.*$^~[]')
13     let l:pattern = substitute(l:pattern, "\n$", "", "")
14 
15     if a:direction == 'b'
16         execute "normal ?" . l:pattern . "^M"
17     elseif a:direction == 'gv'
18         call CmdLine("vimgrep " . '/'. l:pattern . '/' . ' **/*.')
19     elseif a:direction == 'f'
20         execute "normal /" . l:pattern . "^M"
21     endif
22 
23     let @/ = l:pattern
24     let @" = l:saved_reg
25 endfunction

 1 def RemoveBreakpoints():
 2     import re
 3     nCurrentLine = int( vim.eval( 'line(".")'))
 4     nLines = []
 5     nLine = 1
 6     for strLine in vim.current.buffer:
 7         if strLine == 'import ipdb' or strLine.lstrip()[:15] == 'ipdb.set_trace()':
 8             nLines.append( nLine)
 9         nLine += 1
10     nLines.reverse()
11     for nLine in nLines:
12         vim.command( 'normal %dG' % nLine)
13         vim.command( 'normal dd')
14         if nLine < nCurrentLine:
15             nCurrentLine -= 1
16     vim.command( 'normal %dG' % nCurrentLine)
17 EOF
18 map <F2> :py RemoveBreakpoints()<cr>
